Inputs: Text

Text input let users enter and edit single line text.

Please fill in your name.

<template>
  <div class="form">
    <div class="input">
      <SInputText
        v-model="data.name"
        name="name"
        label="Label"
        note="Required"
        help="Please fill in your name."
        placeholder="John Doe"
        :validation="validation.name"
      />
    </div>
    <ul class="actions">
      <li class="action">
        <SButton label="RESET" type="secondary" @click="validation.$reset" />
      </li>
      <li class="action">
        <SButton label="VALIDATE" @click="validation.$validate" />
      </li>
    </ul>
  </div>
</template>

<script lang="ts">
import { defineComponent } from '@nuxtjs/composition-api'
import { required } from '@globalbrain/sefirot/lib/validation/rules'
import useForm from '@globalbrain/sefirot/lib/compositions/useForm'
import SButton from '@globalbrain/sefirot/lib/components/SButton.vue'
import SInputText from '@globalbrain/sefirot/lib/components/SInputText.vue'

export default defineComponent({
  components: {
    SButton,
    SInputText
  },

  setup () {
    const { data, validation } = useForm({
      data: {
        name: ''
      },
      rules: {
        name: [required('Name is required.')]
      }
    })

    return {
      data,
      validation
    }
  }
})
</script>

With Icon

You can supply an icon to the input field by passing in the icon component to the icon prop.

Input with Icon

<template>
  <div class="input">
    <SInputText
      v-model="text"
      name="name"
      label="Label"
      :icon="icon"
      placeholder="Start searching..."
    />
  </div>
</template>

<script lang="ts">
import { defineComponent, ref } from '@nuxtjs/composition-api'
import SIconSearch from '@globalbrain/sefirot/lib/components/icons/SIconSearch.vue'
import SInputText from '@globalbrain/sefirot/lib/components/SInputText.vue'

export default defineComponent({
  components: {
    SInputText
  },

  setup () {
    const text = ref('')

    return {
      icon: SIconSearch,
      text
    }
  }
})
</script>

<style lang="postcss" scoped>
@import "@/assets/styles/variables";

.input {
  max-width: 320px;
}
</style>

Clearable Input

By setting clearable props, the input filed will show the "clear" button when there's any value is set. When the "clear" button is clicked, it will emit the clear event. It's up to you to handle the clear event. The component will not touch the value passed in.

Clearable Input

<template>
  <div class="input">
    <SInputText
      v-model="text"
      name="name"
      label="Label"
      clearable
      placeholder="Start searching..."
      @clear="text = ''"
    />
  </div>
</template>

<script lang="ts">
import { defineComponent, ref } from '@nuxtjs/composition-api'
import SIconSearch from '@globalbrain/sefirot/lib/components/icons/SIconSearch.vue'
import SInputText from '@globalbrain/sefirot/lib/components/SInputText.vue'

export default defineComponent({
  components: {
    SInputText
  },

  setup () {
    const text = ref('Initial value')

    return {
      icon: SIconSearch,
      text
    }
  }
})
</script>
